- query-sort-by:: page
  query-table:: false
  query-sort-desc:: false
  query-properties:: [:page :block :race]
  #+BEGIN_QUERY
  {
  :title [:h3 "Персонажи"]
  :query [
    :find (pull ?page [*])
    :where
      [?b :block/properties ?prop]
      [?b :block/pre-block? true]
      [(get ?prop :template-type) ?r]
      [(= "character" ?r)]
      [?b :block/page ?page]
  ]
  }
  #+END_QUERY
